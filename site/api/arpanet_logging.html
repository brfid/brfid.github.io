129    def _update_stats(self, entry: LogEntry):
130        """Update statistics for a component.
131
132        Args:
133            entry: Log entry to process
134        """
135        component = entry.component
136
137        if component not in self._stats:
138            self._stats[component] = ComponentStats(component=component)
139
140        stats = self._stats[component]
141        stats.total_lines += 1
142
143        # Update log levels
144        if entry.log_level not in stats.log_levels:
145            stats.log_levels[entry.log_level] = 0
146        stats.log_levels[entry.log_level] += 1
147
148        # Update tags
149        for tag in entry.tags:
150            if tag not in stats.tags:
151                stats.tags[tag] = 0
152            stats.tags[tag] += 1
153
154        # Update timestamps
155        if stats.first_timestamp is None:
156            stats.first_timestamp = entry.timestamp
157        stats.last_timestamp = entry.timestamp
158
159        # Count errors/warnings
160        if entry.log_level == "ERROR":
161            stats.errors += 1
162        elif entry.log_level == "WARN":
163            stats.warnings += 1

 63    def start(self):
 64        """Start collecting logs in background thread."""
 65        if self.running:
 66            print(f"  {self.component_name} collector already running")
 67            return
 68
 69        self.running = True
 70        self.thread = threading.Thread(
 71            target=self._collect_loop,
 72            daemon=True,
 73            name=f"collector-{self.component_name}"
 74        )
 75        self.thread.start()
 76        print(f" {self.component_name.upper()} collector started")
 77
 78    def stop(self):
 79        """Stop collecting logs and wait for thread to finish."""
 80        if not self.running:
 81            return
 82
 83        self.running = False
 84        if self.thread:
 85            self.thread.join(timeout=5)
 86        print(f"  {self.component_name.upper()} collector stopped")
 87
 88    def _collect_loop(self):
 89        """Main collection loop (runs in background thread)."""
 90        if self.docker_client is None:
 91            print(f" Docker SDK unavailable, cannot collect {self.component_name} logs")
 92            return
 93
 94        try:
 95            container = self.docker_client.containers.get(self.container_name)
 96
 97            # Stream logs in real-time
 98            for log_line in container.logs(
 99                stream=True,
100                follow=True,
101                timestamps=True
102            ):
103                if not self.running:
104                    break
105
106                try:
107                    # Decode log line
108                    line = log_line.decode('utf-8', errors='replace').rstrip()
109
110                    # Process the line
111                    self._process_line(line)
112
113                except Exception as e:
114                    print(f"Error processing log line from {self.component_name}: {e}")
115                    continue
116
117        except Exception as exc:
118            if docker is not None and isinstance(exc, docker.errors.NotFound):
119                print(f" Container {self.container_name} not found")
120                return
121            print(f" Error in {self.component_name} collector: {exc}")
122
123    def _process_line(self, line: str):
124        """Process a single log line.
125
126        Args:
127            line: Raw log line with timestamp prefix
128        """
129        # Parse Docker timestamp format: "2026-02-07T22:30:15.123456789Z message"
130        try:
131            if ' ' in line:
132                timestamp_str, message = line.split(' ', 1)
